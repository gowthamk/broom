  Func<TIn, TOut> selector;

  void onReceive(Time t, List<TIn> inList) {

    foreach (TIn input in inList) {
      TOut output = selector(input);
      map[t].add(output); } }
  void onNotify(Time t) {
     List<TOut> outList = map[t];
     map.Remove(t);

}
